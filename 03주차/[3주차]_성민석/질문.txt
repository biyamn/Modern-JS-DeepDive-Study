2.ouetr함수 스코프
 var z= "outer's local z"; 값이 수집되고

3. inner함수 스코프
  var x = "inner's local x"; 값이 수집되고

   console.log(x); // inner함수 스코프에서 값을 가져옴
   console.log(y); // inner함수와 outer함수 스코프에 값이 없기에 전역에서 가저온다
   console.log(z); // outer에서 가져온다
